#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>

int pdes[2];

int main(void){
   int pid;
   char buffer[64];
   char *msg_fils="Gnark gnark ! je suis le fils!\n";
   char *msg_pere="Ouf ouf ouf, je suis le père\n";
   pipe(pdes);
   
   
   pid = fork();

//    memset(buffer,0,64);
   
   if (pid < 0){
      printf("Erreur : fork()\n");
      return -1;
   }
   if (pid == 0){
      printf("Je suis le fils (pid %d)\n",getpid());
      read(pdes[0],buffer,strlen(msg_pere)*sizeof(char)+1);
      printf("Mon pere m'envoie ceci : %s\n",buffer);
      write(pdes[1],msg_fils,strlen(msg_fils)*sizeof(char)+1);
   }
   else{
      printf("Je suis le père (pid %d)\n",getpid());
      write(pdes[1],msg_pere,strlen(msg_pere)*sizeof(char)+1);
      wait();
      read(pdes[0],buffer,strlen(msg_fils)*sizeof(char)+1);
      printf("Mon fils m'envoie ceci : %s\n",buffer);
      
   }
   
   close(pdes[0]);
   close(pdes[1]);
   
   return 0;
}

#include <stdio.h>
#include <unistd.h>
#include <signal.h>
#include <string.h>

int main(void){
   
   int pid;
   char str[16];
   
   pid=fork();
   
   if (pid<0){
      printf("erreur : fork !\n");
      return -1;
   }
   if (pid==0){
      for(;;){
	 printf("\033[1;34m Je suis le processus %d\n",getpid());
	 sleep(1);
      }
   }
   else{
      for(;;){
	 printf("\n\t \033[1;31m Je suis le processus %d\n",getpid());
	 fgets(str,16,stdin);
	 if (strstr(str,"kill\n")!=NULL){
	    printf("\n\t \033[1;31m JE TUE MON FILS !\n");
	    if (kill(pid,SIGKILL)!=0)
	       perror("Je n'ai pas réussit"); 
	    return 0;
	 }
      }
   }
   return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/msg.h>

#define TAILLE_MSG 48

typedef struct msg{
   long int mtype;
   char mtext[TAILLE_MSG];
}message;

int main(int argc, char *argv[]){
   
   int msqid;
   key_t clef;
   message resultat;
   int secondes=0;
   int msgflg = IPC_CREAT | 0400;
   
   clef = ftok("/tmp",atoi(argv[0]));
   
   if ((msqid = msgget(clef, msgflg))<0) {
      perror("msgget");
      exit(-1);
   }
   while(1) {
      if ((msgrcv(msqid,&resultat,TAILLE_MSG,1,IPC_NOWAIT))<0) perror("msgrcv");
      else break;
      sleep(1);
      secondes++;
   }
   printf("Voici le résultat : %s\n
	    Le programme à attendu environs %d secondes\n"
	  ,resultat.mtext,secondes);
   
   if ((msgctl(msqid,IPC_RMID,(struct msqid_ds *)NULL))<0) {
      perror("msgctl");
   }
   return 0;
}
